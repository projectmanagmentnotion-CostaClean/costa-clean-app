  20: function buildPropertyAddressLine(property: PropertyListItem | undefined): string | null {
  21:   if (!property) return null
  22: 
  23:   const parts = [
  24:     property.address?.trim(),
  25:     property.postal_code?.trim(),
  26:     property.city?.trim(),

  29:   return parts.length > 0 ? parts.join(', ') : null
  30: }

  32: function buildServiceReference(
  33:   invoice: InvoiceListItem,
  34:   job: JobListItem | undefined,
  35:   property: PropertyListItem | undefined,
  36: ): string | null {
  37:   const parts = [
  38:     invoice.job_display_code ?? job?.display_code ?? invoice.job_id,
  39:     property?.display_code ?? null,
  40:     property?.name ?? null,
  41:   ].filter(Boolean)

  43:   return parts.length > 0 ? parts.join(' · ') : null

  46: function buildServiceDescription(
  47:   job: JobListItem | undefined,
  48:   property: PropertyListItem | undefined,
  49: ): string | null {
  50:   const serviceType = job?.service_type?.trim() || 'Servicio de limpieza'
  51:   const propertyName = property?.name?.trim()
  52:   const propertyType = property?.property_type?.trim()
  53:   const city = property?.city?.trim()
  54: 
  55:   const locationBits = [propertyName, propertyType, city].filter(Boolean)
  56:   return locationBits.length > 0
  57:     ? `${serviceType} en ${locationBits.join(' · ')}`
  58:     : serviceType
  59: }
